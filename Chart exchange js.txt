var feb = 27

function calcDay(dia,mes,ano){
    ano = parseInt(ano)
    if (ano%4 === 0 && ano%100 !== 100){
        feb ++
    }
    switch (mes) {
        case '01':
            dia = parseInt(dia)
            arrfec[dia - 1] ++
            break;
        case '02':
            dia = parseInt(dia)
            arrfec[dia + 30] ++
            break;
        case '03':
            dia = parseInt(dia)
            arrfec[dia + feb + 30] ++
            break;
        case '04':
            dia = parseInt(dia)
            arrfec[dia + feb + 61] = arrfec[dia + feb + 61] + 1
            console.log(dia + feb + 61, arrfec[dia + feb + 61])
            break;
        case '05':
            dia = parseInt(dia)
            arrfec[dia + feb + 91] ++
            break;
        case '06':
            dia = parseInt(dia)    
            arrfec[dia + feb + 122] ++
            console.log(dia + feb + 122, arrfec[dia + feb + 122])
            break;
        case '07':
            dia = parseInt(dia)
            arrfec[dia + feb + 152] ++
            break;
        case '08':
            dia = parseInt(dia)
            arrfec[dia + feb + 183] ++
            break;
        case '09':
            dia = parseInt(dia)
            arrfec[dia + feb + 214] ++
            break;
        case '10':
            dia = parseInt(dia)
            arrfec[dia + feb + 244] ++
            break;
        case '11':
            dia = parseInt(dia)
            arrfec[dia + feb + 275] ++
            break;  
        case '12':
            dia = parseInt(dia)
            arrfec[dia + feb + 305] ++
            break;  
        default:
            break;
    }
}
